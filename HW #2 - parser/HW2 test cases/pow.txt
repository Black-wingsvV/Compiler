/*
	Computes num1 to the power of num2. You need
	to enter two values at the terminal to use this
	case, since it reads in num1 and reads in num2
	toward the beginning of execution.

	For instance, feeding this 5 and 6 should yield
	the result of 5^6 = 15625.
*/

var num1, num2, ans;
begin
	ans := 1;
	read num1;
	read num2;
	
	while num2 > 0 do
	begin
		ans := ans * num1;
		num2 := num2 - 1;
	end;
	write ans
end.
